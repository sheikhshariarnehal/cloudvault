## Key Design Decisions

1. **Hilt for DI** - Standard Android DI, integrates well with Compose and WorkManager
2. **Room for offline cache** - Fast local queries, sync with remote on demand
3. **Retrofit + OkHttp** - Industry standard, excellent coroutine support
4. **Moshi for JSON** - Faster than Gson, Kotlin codegen support
5. **Coil for images** - Native Compose support, efficient caching
6. **WorkManager for background** - Guaranteed execution, survives app restarts
7. **DataStore for preferences** - Modern replacement for SharedPreferences
8. **Guest session first** - Match web app behavior, UUID stored locally

## API Integration Strategy

All API calls go through Retrofit interfaces matching the existing backend:

- **Auth**: Supabase auth endpoints (or direct API if custom)
- **Files**: `/api/files` (GET, POST, PATCH)
- **Folders**: `/api/folders` (GET, POST, PATCH, DELETE)
- **Upload**: `/api/upload` (POST multipart), `/api/upload/init` + `/api/upload/chunk` + `/api/upload/complete` for large files
- **Download**: `/api/download/{id}` (GET, returns signed URL redirect)
- **Share**: `/api/share` (POST, GET)
- **Telegram**: `/api/telegram/status`, `/api/telegram/send-code`, etc.

## Backend URL Configuration

The app will need:
- `BASE_URL` - Backend API base URL (configurable via build config or settings)
- Guest session ID generated client-side, stored in DataStore
- Auth token from Supabase stored securely (EncryptedSharedPreferences)

## State Management

- **ViewModel + StateFlow** for screen state
- **UiState sealed class**: `Loading`, `Success(data)`, `Error(message)`
- **Pagination**: Offset-based for file lists (backend supports it)
- **Optimistic updates**: For star/trash/rename operations

## Error Handling

- Network errors with retry capability
- Offline detection with appropriate UI
- File size validation (2GB limit)
- Rate limit handling (429 with retry-after)

## Security

- HTTPS only (network security config)
- Token storage in EncryptedSharedPreferences
- Certificate pinning (optional, for production)
- No sensitive data in logs

```kotlin
package com.cloudvault.android.util

import java.util.Locale

object FileExtensions {
    private val IMAGE_EXTENSIONS = setOf("jpg", "jpeg", "png", "gif", "webp", "svg", "heic", "bmp")
    private val VIDEO_EXTENSIONS = setOf("mp4", "mov", "avi", "mkv", "webm", "wmv")
    private val AUDIO_EXTENSIONS = setOf("mp3", "wav", "ogg", "flac", "aac", "m4a")
    private val DOCUMENT_EXTENSIONS = setOf("doc", "docx", "xls", "xlsx", "ppt", "pptx", "pdf", "txt", "md", "csv")
    private val ARCHIVE_EXTENSIONS = setOf("zip", "rar", "7z", "tar", "gz")

    enum class FileType {
        IMAGE, VIDEO, AUDIO, DOCUMENT, ARCHIVE, OTHER
    }

    fun getFileType(fileName: String): FileType {
        val ext = getExtension(fileName).lowercase(Locale.ROOT)
        return when {
            ext in IMAGE_EXTENSIONS -> FileType.IMAGE
            ext in VIDEO_EXTENSIONS -> FileType.VIDEO
            ext in AUDIO_EXTENSIONS -> FileType.AUDIO
            ext in DOCUMENT_EXTENSIONS -> FileType.DOCUMENT
            ext in ARCHIVE_EXTENSIONS -> FileType.ARCHIVE
            else -> FileType.OTHER
        }
    }

    fun getExtension(fileName: String): String {
        val lastDot = fileName.lastIndexOf('.')
        return if (lastDot > 0) fileName.substring(lastDot + 1) else ""
    }

    fun getFileNameWithoutExtension(fileName: String): String {
        val lastDot = fileName.lastIndexOf('.')
        return if (lastDot > 0) fileName.substring(0, lastDot) else fileName
    }
}

fun Long.formatFileSize(): String {
    if (this <= 0) return "0 B"
    val units = arrayOf("B", "KB", "MB", "GB", "TB")
    val digitGroups = (Math.log10(toDouble()) / Math.log10(1024.0)).toInt()
    val value = this / Math.pow(1024.0, digitGroups.toDouble())
    return String.format(Locale.US, "%.1f %s", value, units[digitGroups])
}
```

### File 9: `util/NetworkUtils.kt`

```kotlin
package com.cloudvault.android.util

import android.content.Context
import android.net.ConnectivityManager
import android.net.NetworkCapabilities
import dagger.hilt.android.qualifiers.ApplicationContext
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class NetworkUtils @Inject constructor(
    @ApplicationContext private val context: Context
) {
    fun isOnline(): Boolean {
        val connectivityManager = context.getSystemService(
            Context.CONNECTIVITY_SERVICE
        ) as ConnectivityManager
        val network = connectivityManager.activeNetwork ?: return false
        val capabilities = connectivityManager.getNetworkCapabilities(network) ?: return false
        return capabilities.hasCapability(NetworkCapabilities.NET_CAPABILITY_INTERNET)
    }
}
```

```kotlin
    dynamicColor: Boolean = true,

) {

            color = if (darkTheme) BackgroundDark else BackgroundLight,
            darkIcons = !darkTheme
        )
    }

    MaterialTheme(
        colorScheme = colorScheme,
        typography = Typography,
        content = content
    )
}
```

```kotlin
import kotlinx.coroutines.launch
import javax.inject.Inject

data class HomeUiState(
    val files: List<CloudFile> = emptyList(),
    val folders: List<CloudFolder> = emptyList(),
    val isLoading: Boolean = false,
    val error: String? = null,
    val currentFolderId: String? = null,
    val currentFolderName: String = "My Drive",
    val isGridView: Boolean = true,
    val isGuest: Boolean = true,
    val storageUsed: Long = 0,
    val storageLimit: Long = 10L * 1024 * 1024 * 1024 // 10GB default
)

@HiltViewModel
class HomeViewModel @Inject constructor(
    private val fileRepository: FileRepository,
    private val folderRepository: FolderRepository,
    private val authRepository: AuthRepository
) : ViewModel() {
    private val _uiState = MutableStateFlow(HomeUiState(isGuest = authRepository.isGuest()))
    val uiState: StateFlow<HomeUiState> = _uiState.asStateFlow()

    init {
        loadFiles()
        loadFolders()
    }

    fun loadFiles() {
        viewModelScope.launch {
            fileRepository.getFiles(
                userId = if (authRepository.isGuest()) null else authRepository.authToken,
                guestSessionId = if (authRepository.isGuest()) authRepository.guestSessionId else null,
                folderId = _uiState.value.currentFolderId,
                starred = null
            ).collect { resource ->
                when (resource) {
                    is Resource.Loading -> {
                        _uiState.update { it.copy(isLoading = true, error = null) }
                    }
                    is Resource.Success -> {
                        _uiState.update {
                            it.copy(
                                files = resource.data ?: emptyList(),
                                isLoading = false
                            )
                        }
                    }
                    is Resource.Error -> {
                        _uiState.update {
                            it.copy(isLoading = false, error = resource.message)
                        }
                    }
                }
            }
        }
    }

    fun loadFolders() {
        viewModelScope.launch {
            folderRepository.getFolders(
                userId = if (authRepository.isGuest()) null else authRepository.authToken,
                guestSessionId = if (authRepository.isGuest()) authRepository.guestSessionId else null
            ).collect { resource ->
                when (resource) {
                    is Resource.Loading -> {
                        _uiState.update { it.copy(isLoading = true) }
                    }
                    is Resource.Success -> {
                        _uiState.update {
                            it.copy(
                                folders = resource.data ?: emptyList(),
                                isLoading = false
                            )
                        }
                    }
                    is Resource.Error -> {
                        _uiState.update { it.copy(isLoading = false, error = resource.message) }
                    }
                }
            }
        }
    }

    fun navigateToFolder(folderId: String, folderName: String) {
        _uiState.update {
            it.copy(
                currentFolderId = folderId,
                currentFolderName = folderName
            )
        }
        loadFiles()
    }

    fun navigateUp() {
        _uiState.update {
            it.copy(
                currentFolderId = null,
                currentFolderName = "My Drive"
            )
        }
        loadFiles()
    }

    fun toggleViewMode() {
        _uiState.update { it.copy(isGridView = !it.isGridView) }
    }

    fun starFile(fileId: String, isStarred: Boolean) {
        viewModelScope.launch {
            fileRepository.starFile(fileId, !isStarred)
        }
    }

    fun trashFile(fileId: String) {
        viewModelScope.launch {
            fileRepository.trashFile(fileId, true)
        }
    }

    fun refresh() {
        loadFiles()
        loadFolders()
    }
}
```

```kotlin
package com.cloudvault.android.util

import java.util.Locale

object FileExtensions {
    private val IMAGE_EXTENSIONS = setOf("jpg", "jpeg", "png", "gif", "webp", "svg", "heic", "bmp")
    private val VIDEO_EXTENSIONS = setOf("mp4", "mov", "avi", "mkv", "webm", "wmv")
    private val AUDIO_EXTENSIONS = setOf("mp3", "wav", "ogg", "flac", "aac", "m4a")
    private val DOCUMENT_EXTENSIONS = setOf("doc", "docx", "xls", "xlsx", "ppt", "pptx", "pdf", "txt", "md", "csv")
    private val ARCHIVE_EXTENSIONS = setOf("zip", "rar", "7z", "tar", "gz")

    enum class FileType {
        IMAGE, VIDEO, AUDIO, DOCUMENT, ARCHIVE, OTHER
    }

    fun getFileType(fileName: String): FileType {
        val ext = getExtension(fileName).lowercase(Locale.ROOT)
        return when {
            ext in IMAGE_EXTENSIONS -> FileType.IMAGE
            ext in VIDEO_EXTENSIONS -> FileType.VIDEO
            ext in AUDIO_EXTENSIONS -> FileType.AUDIO
            ext in DOCUMENT_EXTENSIONS -> FileType.DOCUMENT
            ext in ARCHIVE_EXTENSIONS -> FileType.ARCHIVE
            else -> FileType.OTHER
        }
    }

    fun getExtension(fileName: String): String {
        val lastDot = fileName.lastIndexOf('.')
        return if (lastDot > 0) fileName.substring(lastDot + 1) else ""
    }

    fun getFileNameWithoutExtension(fileName: String): String {
        val lastDot = fileName.lastIndexOf('.')
        return if (lastDot > 0) fileName.substring(0, lastDot) else fileName
    }
}

fun Long.formatFileSize(): String {
    if (this <= 0) return "0 B"
    val units = arrayOf("B", "KB", "MB", "GB", "TB")
    val digitGroups = (Math.log10(toDouble()) / Math.log10(1024.0)).toInt()
    val value = this / Math.pow(1024.0, digitGroups.toDouble())
    return String.format(Locale.US, "%.1f %s", value, units[digitGroups])
}
```

```kotlin
import android.net.ConnectivityManager
import android.net.NetworkCapabilities
import dagger.hilt.android.qualifiers.ApplicationContext
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class NetworkUtils @Inject constructor(
    @ApplicationContext private val context: Context
) {
    fun isOnline(): Boolean {
        val connectivityManager = context.getSystemService(
            Context.CONNECTIVITY_SERVICE
        ) as ConnectivityManager
        val network = connectivityManager.activeNetwork ?: return false
        val capabilities = connectivityManager.getNetworkCapabilities(network) ?: return false
```

```kotlin
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
import javax.inject.Inject

data class HomeUiState(
    val files: List<CloudFile> = emptyList(),
    val folders: List<CloudFolder> = emptyList(),
    val isLoading: Boolean = false,
    val error: String? = null,
    val currentFolderId: String? = null,
    val currentFolderName: String = "My Drive",
    val isGridView: Boolean = true,
    val isGuest: Boolean = true,
    val storageUsed: Long = 0,
    val storageLimit: Long = 10L * 1024 * 1024 * 1024 // 10GB default
)

@HiltViewModel
class HomeViewModel @Inject constructor(
    private val fileRepository: FileRepository,
    private val folderRepository: FolderRepository,
    private val authRepository: AuthRepository
) : ViewModel() {
    private val _uiState = MutableStateFlow(HomeUiState(isGuest = authRepository.isGuest()))
    val uiState: StateFlow<HomeUiState> = _uiState.asStateFlow()

    init {
        loadFiles()
        loadFolders()
    }

    fun loadFiles() {
        viewModelScope.launch {
            fileRepository.getFiles(
                userId = if (authRepository.isGuest()) null else authRepository.authToken,
                guestSessionId = if (authRepository.isGuest()) authRepository.guestSessionId else null,
                folderId = _uiState.value.currentFolderId,
                starred = null
            ).collect { resource ->
                when (resource) {
                    is Resource.Loading -> {
                        _uiState.update { it.copy(isLoading = true, error = null) }
                    }
                    is Resource.Success -> {
                        _uiState.update {
                            it.copy(
                                files = resource.data ?: emptyList(),
                                isLoading = false
                            )
                        }
                    }
                    is Resource.Error -> {
                        _uiState.update {
                            it.copy(isLoading = false, error = resource.message)
                        }
                    }
                }
            }
        }
    }

    fun loadFolders() {
        viewModelScope.launch {
            folderRepository.getFolders(
                userId = if (authRepository.isGuest()) null else authRepository.authToken,
                guestSessionId = if (authRepository.isGuest()) authRepository.guestSessionId else null
            ).collect { resource ->
                when (resource) {
                    is Resource.Loading -> {
                        _uiState.update { it.copy(isLoading = true) }
                    }
                    is Resource.Success -> {
                        _uiState.update {
                            it.copy(
                                folders = resource.data ?: emptyList(),
                                isLoading = false
                            )
                        }
                    }
                    is Resource.Error -> {
                        _uiState.update { it.copy(isLoading = false, error = resource.message) }
                    }
                }
            }
        }
    }

    fun navigateToFolder(folderId: String, folderName: String) {
        _uiState.update {
            it.copy(
                currentFolderId = folderId,
                currentFolderName = folderName
            )
        }
        loadFiles()
    }

    fun navigateUp() {
        _uiState.update {
            it.copy(
                currentFolderId = null,
                currentFolderName = "My Drive"
            )
        }
        loadFiles()
    }

    fun toggleViewMode() {
        _uiState.update { it.copy(isGridView = !it.isGridView) }
    }

    fun starFile(fileId: String, isStarred: Boolean) {
        viewModelScope.launch {
            fileRepository.starFile(fileId, !isStarred)
        }
    }

    fun trashFile(fileId: String) {
        viewModelScope.launch {
            fileRepository.trashFile(fileId, true)
        }
    }

    fun refresh() {
        loadFiles()
        loadFolders()
    }
}
```
